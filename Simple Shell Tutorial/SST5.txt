#!/bin/bash

# Case


read -p "How old are ya?" age

case $age in
[0-4])
	echo "To Young for school"
	;;

5)
	echo "Go to Kindergarten"
	;;

[6-9]|[0-8])
	grade=$((age-5))
	echo "Go to grade $grade"
	;;

# This is like the default
*)
	echo "You are too old for school"
	;;
esac


# Ternery Operatpr

can_vote=0
age=18
((age >= 18?(can_vote=1):(can_vote)))
echo "Can Vote : $can_vote"
